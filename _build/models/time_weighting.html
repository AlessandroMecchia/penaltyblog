Create the Weights#

[4]:

xi = 0.001
weights = pb.models.dixon_coles_weights(df["date"], xi)

Train the model#

[5]:

clf = pb.models.DixonColesGoalModel(
    df["goals_home"], df["goals_away"], df["team_home"], df["team_away"], weights
)
clf.fit()
